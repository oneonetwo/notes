1. 堆是一个完全二叉树
2. 堆的父节点的值大于子节点
3. 可以用数组来表示一颗完全二叉树
```javascript
function swap(arr, max, i){
    let temp = arr[max];
    arr[max] = arr[i];
    arr[i] = temp;
}

//对某个节点做heapify的操作
function heapify(list, len, i){
  //list表示堆的数组 len是节点的长度。 cur当前的节点   
    console.log('i', i);
  if(i>=len) {
    return list
  };
  //c1 c2分别表示当前子节点的左右节点指针
  let c1 = 2*i+1;
  let c2 = 2*i+2;
  //找出值最大的指针,注意边界
  let max = i;
  if(c1<len && list[c1]>list[max]){
      max = c1;
  }
  if(c2<len && list[c2]>list[max]){
      max = c2;
  }
  if(max!==i){
    //交换两个节点的位置
    swap(list, max, i);
    heapify(list, len, max);
  }
}

//如果整个数组是乱的，我们找见最后一节点的父节点，以此 i-1 做 heapify
var arr1 = [4,10,3,5,1,2]
var arr2 = heapify(arr1, arr1.length,0)
```
